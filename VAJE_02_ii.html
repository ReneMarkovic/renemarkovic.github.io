	print("Ime vozlia = ",i,"povezanost vozlia = ",G.degree(i))
	avg+=G.degree(i)
print("<k> = ", avg/len(G))

#BARABASI
G=nx.barabasi_albert_graph(40, 2)
#pos=nx.spring_layout(G,iterations=100)
pos=nx.random_layout(G)

k=np.zeros([len(G)],int)

for i in G:
	k[i]=G.degree(i)

plt.subplot(221)
nx.draw(G,pos,font_size=8,node_size=k*10.0)

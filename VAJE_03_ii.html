import numpy as np
import matplotlib.pyplot as plt

M=5.97*pow(10,24)
R=6400000.0

def sign(a):
    xx=a
    if abs(xx)>0.0:
        return xx/abs(xx)
    else:
        return 1.0

def gravity(M,R):
    G=6.67*pow(10.0,-11.0)
    g=(G*M/R**2)
    return g

def posp(v,S,c,m,g):
    fi=np.arctan(v[1]/v[0])
    fi=abs(fi)
    abs_v=np.sqrt(v.dot(v))
    Fu=0.5*1.3*S*c*abs_v
    ax=-sign(v[0])*Fu*np.cos(fi)/m
    ay=-g-sign(v[1])*Fu*np.sin(fi)/m
    az=0.0
    return np.array([ax,ay,az])
